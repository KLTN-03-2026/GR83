-- 1) Bảng Quyền
CREATE TABLE dbo.Quyen
(
    MaQuyen   CHAR(2)       NOT NULL,
    TenQuyen  NVARCHAR(30)  NOT NULL,
    MoTa      NVARCHAR(100) NOT NULL,

    CONSTRAINT PK_Quyen PRIMARY KEY (MaQuyen),
    CONSTRAINT UQ_Quyen_Ten UNIQUE (TenQuyen),
    CONSTRAINT CK_Quyen_Ma CHECK (MaQuyen IN ('Q1','Q2','Q3')),
    CONSTRAINT CK_Quyen_Ten CHECK (TenQuyen IN (N'QuanTriVien', N'KhachHang', N'TaiXe'))
);
GO

-- 2) Bảng TàiKhoản
CREATE TABLE dbo.TaiKhoan
(
    MaTK         VARCHAR(20)   NOT NULL,
    TaiKhoan     VARCHAR(150)  NOT NULL,
    MatKhau      VARCHAR(255)  NOT NULL,
    MaQuyen      CHAR(2)       NOT NULL,
    Ten          NVARCHAR(100) NOT NULL,
    Email        VARCHAR(150)  NOT NULL,
    SDT          VARCHAR(15)   NULL,
    DiaChi       NVARCHAR(255) NULL,
    Avatar       NVARCHAR(500) NULL,
    NgaySinh     DATE          NULL,
    GioiTinh     NVARCHAR(10)  NULL CONSTRAINT DF_TaiKhoan_GioiTinh DEFAULT N'Khac',
    NgayTao      DATETIME2(0)  NOT NULL CONSTRAINT DF_TaiKhoan_NgayTao DEFAULT SYSDATETIME(),
    NgayCapNhat  DATETIME2(0)  NOT NULL CONSTRAINT DF_TaiKhoan_NgayCapNhat DEFAULT SYSDATETIME(),
    TrangThai    NVARCHAR(20)  NOT NULL CONSTRAINT DF_TaiKhoan_TrangThai DEFAULT N'HoatDong',

    CONSTRAINT PK_TaiKhoan PRIMARY KEY (MaTK),
    CONSTRAINT UQ_TaiKhoan_TaiKhoan UNIQUE (TaiKhoan),
    CONSTRAINT UQ_TaiKhoan_Email UNIQUE (Email),
    CONSTRAINT UQ_TaiKhoan_SDT UNIQUE (SDT),
    CONSTRAINT CK_TaiKhoan_GioiTinh CHECK (GioiTinh IN (N'Nam', N'Nu', N'Khac')),
    CONSTRAINT CK_TaiKhoan_TrangThai CHECK (TrangThai IN (N'HoatDong', N'Khoa')),
    CONSTRAINT FK_TaiKhoan_Quyen FOREIGN KEY (MaQuyen)
        REFERENCES dbo.Quyen(MaQuyen)
        ON UPDATE CASCADE
        ON DELETE NO ACTION
);
GO

-- Chỉ có tối đa 1 tài khoản Admin (Q1)
CREATE UNIQUE INDEX UX_TaiKhoan_OnlyOneAdmin
ON dbo.TaiKhoan (MaQuyen)
WHERE MaQuyen = 'Q1';
GO

-- Trigger cập nhật NgayCapNhat khi UPDATE
CREATE OR ALTER TRIGGER dbo.TR_TaiKhoan_SetNgayCapNhat
ON dbo.TaiKhoan
AFTER UPDATE
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE tk
    SET NgayCapNhat = SYSDATETIME()
    FROM dbo.TaiKhoan tk
    INNER JOIN inserted i ON i.MaTK = tk.MaTK;
END
GO

-- 3) Bảng Ngân Hàng

-- 4) Bảng Tài Xế
CREATE TABLE dbo.TaiXe
(
    MaTK         VARCHAR(20)    NOT NULL,
    Avatar       NVARCHAR(500)  NULL,
    DiaChi       NVARCHAR(255)  NULL,
    CCCD         VARCHAR(20)    NOT NULL,
    BangLai      NVARCHAR(500)  NULL,
    LyLich       NVARCHAR(500)  NULL,
    MaNH         VARCHAR(20)    NOT NULL,
    ThongTinXe   NVARCHAR(MAX)  NULL,
    LienHeKC     NVARCHAR(MAX)  NULL,
    TrangThai    NVARCHAR(20)   NOT NULL CONSTRAINT DF_TaiXe_TrangThai DEFAULT N'ChoDuyet',
    NgayTao      DATETIME2(0)   NOT NULL CONSTRAINT DF_TaiXe_NgayTao DEFAULT SYSDATETIME(),
    NgayCapNhat  DATETIME2(0)   NOT NULL CONSTRAINT DF_TaiXe_NgayCapNhat DEFAULT SYSDATETIME(),

    CONSTRAINT PK_TaiXe PRIMARY KEY (CCCD),
    CONSTRAINT UQ_TaiXe_MaTK UNIQUE (MaTK),
    CONSTRAINT UQ_TaiXe_MaNH UNIQUE (MaNH),
    CONSTRAINT CK_TaiXe_TrangThai CHECK (TrangThai IN (N'ChoDuyet', N'HoatDong', N'Khoa')),
    CONSTRAINT CK_TaiXe_ThongTinXe_JSON CHECK (ThongTinXe IS NULL OR ISJSON(ThongTinXe) = 1),
    CONSTRAINT CK_TaiXe_LienHeKC_JSON CHECK (LienHeKC IS NULL OR ISJSON(LienHeKC) = 1),
    CONSTRAINT FK_TaiXe_TaiKhoan FOREIGN KEY (MaTK)
        REFERENCES dbo.TaiKhoan(MaTK)
        ON UPDATE CASCADE
        ON DELETE NO ACTION,
    CONSTRAINT FK_TaiXe_NganHang FOREIGN KEY (MaNH)
        REFERENCES dbo.NganHang(MaNH)
        ON UPDATE CASCADE
        ON DELETE NO ACTION
);
GO

-- Trigger cập nhật NgayCapNhat khi UPDATE bảng TaiXe
CREATE OR ALTER TRIGGER dbo.TR_TaiXe_SetNgayCapNhat
ON dbo.TaiXe

BEGIN
    SET NOCOUNT ON;

    UPDATE tx
    SET NgayCapNhat = SYSDATETIME()
    FROM dbo.TaiXe tx
    INNER JOIN inserted i ON i.MaTK = tx.MaTK;
END
GO

-- 5) Bảng Đặt Xe
CREATE TABLE dbo.DatXe
(
    MaChuyen               VARCHAR(30)    NOT NULL,
    MaTK                   VARCHAR(20)    NULL,
    TenKhachHang           NVARCHAR(100)  NOT NULL,
    SoDienThoai            VARCHAR(15)    NULL,
    LoaiXe                 VARCHAR(20)    NOT NULL,
    DatLich                BIT            NOT NULL CONSTRAINT DF_DatXe_DatLich DEFAULT 0,
    DiemDon                NVARCHAR(255)  NOT NULL,
    DiemDen                NVARCHAR(255)  NOT NULL,
    QuangDuongKm           DECIMAL(10,2)  NOT NULL,
    NguonTuyenDuong        NVARCHAR(30)   NOT NULL,
    MaHangXe               VARCHAR(64)    NOT NULL,
    TenHangXe              NVARCHAR(100)  NOT NULL,
    LoaiGhe                NVARCHAR(50)   NULL,
    ThoiGianDuKienPhut     INT            NOT NULL,
    GiaTien                INT            NOT NULL,
    PhuongThucThanhToan    VARCHAR(20)    NOT NULL,
    NhaCungCapThanhToan    VARCHAR(20)    NULL,
    TrangThaiThanhToan     NVARCHAR(20)   NOT NULL CONSTRAINT DF_DatXe_TrangThaiThanhToan DEFAULT N'ChoXacNhan',
    NgayTao                DATETIME2(0)   NOT NULL CONSTRAINT DF_DatXe_NgayTao DEFAULT SYSDATETIME(),
    NgayCapNhat            DATETIME2(0)   NOT NULL CONSTRAINT DF_DatXe_NgayCapNhat DEFAULT SYSDATETIME(),

    CONSTRAINT PK_DatXe PRIMARY KEY (MaChuyen),
    CONSTRAINT CK_DatXe_LoaiXe CHECK (LoaiXe IN ('motorbike', 'car', 'intercity')),
    CONSTRAINT CK_DatXe_PhuongThucThanhToan CHECK (PhuongThucThanhToan IN ('cash', 'qr', 'wallet')),
    CONSTRAINT CK_DatXe_NhaCungCapThanhToan CHECK (NhaCungCapThanhToan IS NULL OR NhaCungCapThanhToan IN ('zalopay', 'momo', 'shopeepay')),
    CONSTRAINT CK_DatXe_TrangThaiThanhToan CHECK (TrangThaiThanhToan IN (N'ChoThuTien', N'ChoXacNhan', N'DaThanhToan', N'ThatBai')),
    CONSTRAINT FK_DatXe_TaiKhoan FOREIGN KEY (MaTK)
        REFERENCES dbo.TaiKhoan(MaTK)
        ON UPDATE CASCADE
        ON DELETE SET NULL
);
GO
