-- 1.5) DROP ALL FOREIGN KEYS (để tránh lỗi không drop được bảng cha)
DECLARE @DropForeignKeysSql NVARCHAR(MAX) = N'';

SELECT @DropForeignKeysSql = @DropForeignKeysSql +
    N'ALTER TABLE ' + QUOTENAME(OBJECT_SCHEMA_NAME(parent_object_id)) + N'.' + QUOTENAME(OBJECT_NAME(parent_object_id)) +
    N' DROP CONSTRAINT ' + QUOTENAME(name) + N';' + CHAR(10)
FROM sys.foreign_keys;

IF LEN(@DropForeignKeysSql) > 0
BEGIN
    EXEC sp_executesql @DropForeignKeysSql;
END
GO

-- 1) Quyen — bảng tham chiếu, không có FK
CREATE TABLE dbo.Quyen
(
    MaQuyen   CHAR(2)       NOT NULL,
    TenQuyen  NVARCHAR(30)  NOT NULL,
    MoTa      NVARCHAR(100) NOT NULL,

    CONSTRAINT PK_Quyen     PRIMARY KEY (MaQuyen),
    CONSTRAINT UQ_Quyen_Ten UNIQUE (TenQuyen),
    CONSTRAINT CK_Quyen_Ma  CHECK (MaQuyen  IN ('Q1','Q2','Q3')),

-- 2) TaiKhoan
--    MaQuyen (cột thường, FK) → Quyen.MaQuyen (PK)
CREATE TABLE dbo.TaiKhoan
(
    MaTK         VARCHAR(20)   NOT NULL,
    TaiKhoan     VARCHAR(150)  NOT NULL,
    MatKhau      VARCHAR(255)  NOT NULL,
    MaQuyen      CHAR(2)       NOT NULL,              -- FK → Quyen.MaQuyen
    Ten          NVARCHAR(100) NOT NULL,
    Email        VARCHAR(150)  NOT NULL,
    SDT          VARCHAR(15)   NULL,
    DiaChi       NVARCHAR(255) NULL,
    Avatar       NVARCHAR(500) NULL,
    NgaySinh     DATE          NULL,
    GioiTinh     NVARCHAR(10)  NULL     CONSTRAINT DF_TaiKhoan_GioiTinh    DEFAULT N'Khac',
    NgayTao      DATETIME2(0)  NOT NULL CONSTRAINT DF_TaiKhoan_NgayTao     DEFAULT SYSDATETIME(),
    NgayCapNhat  DATETIME2(0)  NOT NULL CONSTRAINT DF_TaiKhoan_NgayCapNhat DEFAULT SYSDATETIME(),
    TrangThai    NVARCHAR(20)  NOT NULL CONSTRAINT DF_TaiKhoan_TrangThai   DEFAULT N'HoatDong',

    CONSTRAINT PK_TaiKhoan           PRIMARY KEY (MaTK),
    CONSTRAINT UQ_TaiKhoan_TaiKhoan  UNIQUE (TaiKhoan),
    CONSTRAINT UQ_TaiKhoan_Email     UNIQUE (Email),
    CONSTRAINT UQ_TaiKhoan_SDT       UNIQUE (SDT),
    CONSTRAINT CK_TaiKhoan_GioiTinh  CHECK (GioiTinh  IN (N'Nam', N'Nu', N'Khac')),
    CONSTRAINT CK_TaiKhoan_TrangThai CHECK (TrangThai IN (N'HoatDong', N'Khoa')),
    CONSTRAINT FK_TaiKhoan_Quyen     FOREIGN KEY (MaQuyen)
        REFERENCES dbo.Quyen(MaQuyen)
        ON UPDATE CASCADE ON DELETE NO ACTION
);
GO

-- Chỉ có tối đa 1 tài khoản Admin (Q1)

    SELECT 1
    FROM sys.indexes
    WHERE name = N'UX_TaiKhoan_OnlyOneAdmin'
      AND object_id = OBJECT_ID(N'dbo.TaiKhoan')
)
BEGIN
    CREATE UNIQUE INDEX UX_TaiKhoan_OnlyOneAdmin ON dbo.TaiKhoan (MaQuyen) WHERE MaQuyen = 'Q1';
END
GO

CREATE OR ALTER TRIGGER dbo.TR_TaiKhoan_SetNgayCapNhat
ON dbo.TaiKhoan AFTER UPDATE AS
BEGIN
    SET NOCOUNT ON;
    UPDATE tk SET tk.NgayCapNhat = SYSDATETIME()
    FROM dbo.TaiKhoan AS tk
    WHERE EXISTS (SELECT 1 FROM inserted AS i WHERE i.MaTK = tk.MaTK);
END
GO

-- 4) TaiXe
--    MaTK (cột thường, UQ, FK) → TaiKhoan.MaTK (PK)
--    MaNH (cột thường, UQ, FK) → NganHang.MaNH (PK)
--    PK của bảng này là CCCD
CREATE TABLE dbo.TaiXe
(
    MaTK         VARCHAR(20)    NOT NULL,              -- FK → TaiKhoan.MaTK
    Avatar       NVARCHAR(500)  NULL,
    DiaChi       NVARCHAR(255)  NULL,
    CCCD         VARCHAR(20)    NOT NULL,              -- PK
    BangLai      NVARCHAR(500)  NULL,
    LyLich       NVARCHAR(500)  NULL,
    MaNH         VARCHAR(20)    NOT NULL,              -- FK → NganHang.MaNH
    ThongTinXe   NVARCHAR(MAX)  NULL,
    LienHeKC     NVARCHAR(MAX)  NULL,
    TrangThai    NVARCHAR(20)   NOT NULL CONSTRAINT DF_TaiXe_TrangThai   DEFAULT N'ChoDuyet',
    KhoaTamDen   DATETIME2(0)   NULL,
    LyDoKhoaTam  NVARCHAR(500)  NULL,
    NgayTao      DATETIME2(0)   NOT NULL CONSTRAINT DF_TaiXe_NgayTao     DEFAULT SYSDATETIME(),
    NgayCapNhat  DATETIME2(0)   NOT NULL CONSTRAINT DF_TaiXe_NgayCapNhat DEFAULT SYSDATETIME(),

    CONSTRAINT PK_TaiXe      PRIMARY KEY (CCCD),
    CONSTRAINT UQ_TaiXe_MaTK UNIQUE (MaTK),
    CONSTRAINT UQ_TaiXe_MaNH UNIQUE (MaNH),
    CONSTRAINT CK_TaiXe_TrangThai       CHECK (TrangThai  IN (N'ChoDuyet', N'HoatDong', N'Khoa')),
    CONSTRAINT CK_TaiXe_ThongTinXe_JSON CHECK (ThongTinXe IS NULL OR ISJSON(ThongTinXe) = 1),
    CONSTRAINT CK_TaiXe_LienHeKC_JSON  CHECK (LienHeKC   IS NULL OR ISJSON(LienHeKC)   = 1),
    CONSTRAINT FK_TaiXe_TaiKhoan FOREIGN KEY (MaTK)
        REFERENCES dbo.TaiKhoan(MaTK)
        ON UPDATE CASCADE ON DELETE NO ACTION,
    CONSTRAINT FK_TaiXe_NganHang FOREIGN KEY (MaNH)
        REFERENCES dbo.NganHang(MaNH)
        ON UPDATE CASCADE ON DELETE NO ACTION
);
GO

-- 4.2) GiaoDichVi — Lịch sử giao dịch ví (dùng chung)
--    MaTK (cột thường, FK) → TaiKhoan.MaTK (PK)
CREATE TABLE dbo.GiaoDichVi
(
    MaGD          INT            IDENTITY(1,1) NOT NULL,
    MaTK          VARCHAR(20)    NOT NULL,              -- FK → TaiKhoan.MaTK
    LoaiGiaoDich  VARCHAR(20)    NOT NULL,
    SoTien        INT            NOT NULL,
    SoDuTruoc     INT            NOT NULL,
    SoDuSau       INT            NOT NULL,
    MoTa          NVARCHAR(255)  NULL,
    SoDTNguoiNhan VARCHAR(20)    NULL,
    SoDTNguoiGui  VARCHAR(20)    NULL,
    MaThamChieu   VARCHAR(40)    NULL,
    TrangThai     VARCHAR(20)    NOT NULL CONSTRAINT DF_GiaoDichVi_TrangThai DEFAULT 'completed',
    NgayTao       DATETIME2(0)   NOT NULL CONSTRAINT DF_GiaoDichVi_NgayTao  DEFAULT SYSDATETIME(),

    CONSTRAINT PK_GiaoDichVi             PRIMARY KEY (MaGD),
    CONSTRAINT CK_GiaoDichVi_Loai        CHECK (LoaiGiaoDich IN ('topup','transfer','receive','adjustment')),
    CONSTRAINT CK_GiaoDichVi_TrangThai   CHECK (TrangThai    IN ('completed','pending','failed')),
    CONSTRAINT CK_GiaoDichVi_SoTien      CHECK (SoTien   <> 0),
    CONSTRAINT CK_GiaoDichVi_SoDuTruoc   CHECK (SoDuTruoc >= 0),
    CONSTRAINT CK_GiaoDichVi_SoDuSau     CHECK (SoDuSau  >= 0),
    CONSTRAINT FK_GiaoDichVi_TaiKhoan    FOREIGN KEY (MaTK)

-- 5) UuDai — Bảng ưu đãi/mã giảm giá (đặt trước DatXe để DatXe khai báo FK)

    CONSTRAINT CK_UuDai_DonToiThieu     CHECK (DonToiThieu >= 0),
    CONSTRAINT CK_UuDai_DoiTuong        CHECK (DoiTuong     IN ('all','customer','driver','admin')),
    CONSTRAINT CK_UuDai_HienThi         CHECK (HienThi      IN ('public','hidden')),
    CONSTRAINT CK_UuDai_TrangThai       CHECK (TrangThai    IN ('active','expired','scheduled')),

CREATE OR ALTER TRIGGER dbo.TR_ViPhamTaiXe_SetNgayCapNhat
ON dbo.ViPhamTaiXe AFTER UPDATE AS
BEGIN
    SET NOCOUNT ON;
    UPDATE vp SET vp.NgayCapNhat = SYSDATETIME()
    FROM dbo.ViPhamTaiXe AS vp
    WHERE EXISTS (SELECT 1 FROM inserted AS i WHERE i.MaVP = vp.MaVP);
END
GO
